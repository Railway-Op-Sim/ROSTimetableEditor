102 QString ROSService::_start_new()
103 {
104  QString out_string = join(";", _service_id,_description, QString(_start_speed), QString(_max_speed),
105  QString(_mass), QString(_max_brake), QString(_power));
106 
107  switch (_service_type)
108  {
109  case ROSService::ServiceType::ServiceFromSplit:
110  out_string = join(",", out_string, join(";", _enter_map_time.toString("HH:mm"), "Sfs", _parent_service));
111  break;
112  case ROSService::ServiceType::ServiceFromService:
113  out_string = join(",", out_string, join(";", _enter_map_time.toString("HH:mm"), "Sns", _parent_service));
114  break;
115  case ROSService::ServiceType::ShuttleFinishService:
116  out_string = join(",", out_string, join(";", _enter_map_time.toString("HH:mm"), "Sns-fsh", _parent_service));
117  break;
118  case ROSService::ServiceType::ShuttleFromStop:
119  out_string = join(",", out_string, join(";", _enter_map_time.toString("HH:mm"), "Snt-sh", _enter_ids[0]));
120  out_string = join(" ", out_string, join(";", _enter_ids[1], _parent_service));
121  break;
122  case ROSService::ServiceType::ShuttleFromFeeder:
123  out_string = join(",", out_string, join(";", _enter_map_time.toString("HH:mm"), "Sns-sh", _parent_service));
124  break;
125  default:
126  out_string = join(",", out_string, join(";", _enter_map_time.toString("HH:mm"), "Snt", _enter_ids[0], _enter_ids[1]));
127  }
128 
129  return out_string;
130 }
