46 QString CloneDialog::_create_new_id(const QString& current_id)
47 {
48  QString _new_id = _current_srv->getID();
49 
50  if(current_id[3].isNumber())
51  {
52  if(current_id[2].isNumber())
53  {
54  int _end_num = current_id.right(2).toInt()+1;
55  while(_current_service_list.contains(current_id.left(2)+QString::number(_end_num)))
56  {
57  _end_num++;
58  if(_end_num > 99) return _new_id;
59  }
60 
61  QString _as_str = (_end_num < 10) ? "0"+QString::number(_end_num) : QString::number(_end_num);
62  _new_id = current_id.left(2)+_as_str;
63  }
64 
65  else
66  {
67  int _end_num = current_id.right(1).toInt()+1;
68  while(_current_service_list.contains(current_id.left(3)+QString::number(_end_num)))
69  {
70  _end_num++;
71  if(_end_num > 9) _new_id = _current_srv->getID();
72  }
73 
74  _new_id = current_id.left(3)+QString::number(_end_num);
75  }
76 
77  }
78 
79  return _new_id;
80 }

92 bool CloneDialog::_check_new()
93 {
94  if(!_current_srv || !_current_ttb)
95  {
96  qDebug() << "Failed to record current service/timetable selection during cloning";
97  return false;
98  }
99  if(_current_srv->getParent() != "")
100  {
101  ui->labelCloneInfo->setText("Currently cannot clone a daughter service.");
102  qDebug() << "Attempt by user to clone a daughter service. Due to the fact parent services would also have to be cloned "\
103  "adding complication, this feature is not yet present.";
104  return false;
105  }
106  if(ui->timeEditCloneStart->time() == _current_srv->getStartTime())
107  {
108  ui->labelCloneInfo->setText("New service start time cannot match old.");
109  qDebug() << "Attempt by user to set clone to be at the same time as the current service.";
110  return false;
111  }
112 
113  if(ui->textEditCloneRef->text() == _current_srv->getID())
114  {
115  ui->labelCloneInfo->setText("New service ID cannot match old.");
116  qDebug() << "Attempt by user to set clone to be have the same ID as the current service.";
117  return false;
118  }
119 
120  return true;
121 }
