61 bool Station_add::setInfo()
62 {
63  if(_srv_start.msecsTo(ui->timeEditArrival->time()) < 0)
64  {
65  ui->errorLabel->setText("Arrival time cannot be before service start");
66  return false;
67  }
68  if(!_edit_mode && ui->comboBoxStations->currentText() != _current_station)
69  {
70  for(auto time_pair : _current_srv->getTimes())
71  {
72  if((_times[0] >= time_pair[0] && _times[0] <= time_pair[1]) || (_times[1] >= time_pair[0] && _times[1] <= time_pair[1]))
73  {
74  ui->errorLabel->setText("Times overlap with current timetable entries");
75  return false;
76  }
77  }
78  }
79  if(ui->timeEditArrival->time().msecsTo(ui->timeEditDeparture->time()) < 0)
80  {
81  ui->errorLabel->setText("Departure time cannot be before arrival.");
82  return false;
83  }
84  ui->errorLabel->setText("");
85 
86  _times[0] = (_stop_class != StopType::StartPoint) ? ui->timeEditArrival->time() : QTime();
87  _times[1] = (_stop_class != StopType::Terminus) ? ui->timeEditDeparture->time() : QTime();
88 
89  if(ui->checkBoxCDT->isChecked())
90  {
91  if(ui->timeEditCDT->time().msecsTo(ui->timeEditArrival->time()) > 0)
92  {
93  ui->errorLabel->setText("Direction Change Time cannot proceed arrival time");
94  return false;
95  }
96 
97  }
98 
99  return true;
100 }

113 void Station_add::on_buttonBoxAddStation_accepted()
114 {
115  _times = {ui->timeEditArrival->time(), ui->timeEditDeparture->time()};
116 
117  if(setInfo())
118  {
119  this->close();
120 
121  QTime _cdt_time = (ui->checkBoxCDT->isChecked()) ? ui->timeEditCDT->time() : QTime();
122 
123  if(_current_srv->getStations().contains(ui->comboBoxStations->currentText()) || (_current_srv->getStations().size() > 0 && _times == _current_srv->getTimes()[_current_srv->getStations().indexOf(_current_station)]))
124  {
125  QString _new_station = ui->comboBoxStations->currentText();
126  _current_srv->updateStation(_current_station, _times, ui->checkBoxCDT->isChecked(), ui->checkBoxPASS->isChecked(), _cdt_time, _new_station);
127  _current_station = _new_station;
128 
129  this->close();
130 
131  reset_state();
132 
133  }
134 
135  else
136  {
137  _current_station = ui->comboBoxStations->currentText();
138  _current_srv->addStation(_times, _current_station);
139  _current_srv->setStopAsPassPoint(_current_srv->getStations().size()-1, ui->checkBoxPASS->isChecked());
140  _current_srv->setDirectionChangeAtStop(_current_srv->getStations().size()-1, ui->checkBoxCDT->isChecked(), _cdt_time);
141  }
142 
143  _redraw_table();
144 
145  if(ui->checkBoxSplit->isChecked())
146  {
147  _current_srv->setSplitData((ui->radioButtonSplitForward->isChecked()) ? QString("F") : QString("R"), _current_srv->getID(), _current_station, ui->timeEditSplit->time().toString("HH:mm"));
148  }
149 
150  else if(ui->checkBoxJoinOther->isChecked())
151  {
152  _current_srv->setJoinData(ui->textEditSplitRef->text(), ui->comboBoxStations->currentText(), ui->timeEditDeparture->time().toString("HH:mm"));
153  }
154 
155 
156  // Ensure the exit time is always beyond the latest station addition
157  if(_current_srv->getExitTime() == QTime() || ui->timeEditArrival->time().secsTo(_current_srv->getExitTime()) < 0) _current_srv->setExitTime(ui->timeEditArrival->time());
158 
159  _service_table->sortItems(0);
160 
161  reset_state();
162  }
163 
164  else
165  {
166  this->show();
167  }
168 }

170 void Station_add::_redraw_table()
171 {
172  _service_table->setRowCount(0);
173 
174  for(int i{0}; i < _current_srv->getStations().size(); ++i)
175  {
176  QTime _arrival_time = _current_srv->getTimes()[i][0],
177  _depart_time = _current_srv->getTimes()[i][1];
178  QStringList _station_item = {_arrival_time.toString("HH:mm"), _depart_time.toString("HH:mm"), _current_srv->getStations()[i]};
179  _service_table->insertRow(_service_table->rowCount());
180  for(int j{0}; j < _station_item.size(); ++j)
181  {
182  QTableWidgetItem* _new_time_item = new QTableWidgetItem(_station_item[j], 0);
183  _new_time_item->setFlags(Qt::ItemIsSelectable|Qt::ItemIsEnabled);
184  _service_table->setItem(_service_table->rowCount()-1, j, _new_time_item);
185  }
186  }
